# 插入排序

插入排序是稳定的，时间复杂度O(n*n)，空间复杂度O(1)

# 实现

function sort(arr) {
  for (let i = 1, l = arr.length; i < l; i++) {
    for (let j = i - 1; j >= 0 && arr[j] > arr[j + 1]; j--) {
      var temp = arr[j]
      arr[j] = arr[j + 1]
      arr[j + 1] = temp
    }
  }
}

var arr = [5, 4, 3, 2, 1]
sort(arr)
console.log(arr)
